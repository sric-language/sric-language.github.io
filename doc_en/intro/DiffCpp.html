Class

C++

#include <math.h>

class Point {
public:
    int x;
    int y;
    double dis(const Point &t) const {
        int dx = t.x - x;
        int dy = t.y - y;
        return sqrt(dx*dx + dy*dy);
    }
};

Sric:

import cstd::*;

struct Point {
    var x: Int;
    var y: Int;
    fun dis(t: & const Point) const: Float {
        var dx = t.x - x;
        var dy = t.y - y;
        return sqrt(dx*dx + dy*dy);
    }
};
